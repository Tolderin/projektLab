@startuml
title SD-17: Zúzalékszóró fej takarítása
hide footbox
skinparam maxMessageSize 250

participant "sp : SnowPlow" as sp
participant "head : GravelHead" as head
participant "l : Lane" as l

sp -> head : clean(l)
activate head

alt [ Van elég zúzalék a tartályban ]
    head -> l : addGravel(amount)
    activate l
    
    l -> l : gravelDepth += amount
    
    note right of l
        a zúzalék megszünteti a jég csúszósságát,
        amíg a később ráhulló hó be nem fedi
    end note
    
    l --> head
    deactivate l
    
    head -> head : fuelAmount -= szükséges_mennyiség
    
else [ Nincs elég zúzalék ]
    note over head, l
        nincs zúzalék, clean() nem hat
    end note
end

head --> sp
deactivate head

@enduml
